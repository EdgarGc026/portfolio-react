import React, { useState, useEffect } from "react"
import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs"
import About from "."

<Meta
  title="Basics/CxAbout/About"
  component={About}
  argTypes={{
    description: {
      name: "Description",
      control: { type: "text" }
    }
  }}
/>

<ArgsTable of={About} />

<Canvas>
  <Story name="About">
    {(args) => {
      const [about, setAbout] = useState(null)
      useEffect(() => {
        fetch(`/aboutComponent`)
        .then(response => response.json())
        .then((data) => setAbout(data[0].description))
      }, [])
      if(!about) {
        return <></>
      }
      return(
        <div className="p-2">
          <About description={args.description || about} />
        </div>
      )
    }}
  </Story>
</Canvas>
